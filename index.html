One can generate a labeled tree's Prüfer sequence by iteratively removing vertices from the tree until only two vertices remain. Specifically, consider a labeled tree T with vertices {1, 2, ..., n}. At step i, remove the leaf with the smallest label and set the ith element of the Prüfer sequence to be the label of this leaf's neighbour. The Prüfer sequence of a labeled tree is unique and has length n − 2. Both coding and decoding can be reduced to integer radix sorting and parallelized.

                    vector> adj;

                    vector pruefer_code() {
                        int n = adj.size();
                        set leafs;
                        vector degree(n);
                        vector killed(n, false);
                        for (int i = 0; i < n; i++) {
                            degree[i] = adj[i].size();
                            if (degree[i] == 1)
                                leafs.insert(i);
                        }
                    
                        vector code(n - 2);
                        for (int i = 0; i < n - 2; i++) {
                            int leaf = *leafs.begin();
                            leafs.erase(leafs.begin());
                            killed[leaf] = true;
                    
                            int v;
                            for (int u : adj[leaf]) {
                                if (!killed[u])
                                    v = u;
                            }
                    
                            code[i] = v;
                            if (--degree[v] == 1)
                                leafs.insert(v);
                        }
                    
                        return code;
                    }
